   @Test
    @Order(3)
    public void test_getEmployeeByName() {
        List<Employee> allEmployees = new ArrayList<Employee>();

        allEmployees.add(new Employee(1, "Arjun Sharma", 47550.00));
        allEmployees.add(new Employee(2, "Sonu Rama", 27250.00));

        String employeeName = "Arjun Sharma";

        when(employeeRepository.findByName(employeeName)).thenReturn(java.util.Optional.ofNullable(allEmployees.get(0))); // Mocking

        String expectedName = "Arjun Sharma";
        String actualName = employeeService.findEmployeeByName(expectedName).getName();

//        assertEquals(expectedId, actualId);
        assertThat(actualName).isEqualTo(expectedName);
    }
